   

  

\doxysection*{An extension to integrate Doxygen© in Unity© Projects.}

Doxygen© Generator Window add possibility to generate Header documentation folder form directories in Unity©. It\textquotesingle{}s supporting multi-\/directories parameters.

\doxysection*{Installation}

\doxysubsection*{Checkout}


\begin{DoxyItemize}
\item go to your unity project\textquotesingle{}s directory
\end{DoxyItemize}


\begin{DoxyCode}{0}
\DoxyCodeLine{cd <your\_project>}
\end{DoxyCode}



\begin{DoxyItemize}
\item checkout submodule Net\+Worked\+Data
\end{DoxyItemize}


\begin{DoxyCode}{0}
\DoxyCodeLine{cd <your\_project\_path>/Assets}
\DoxyCodeLine{git submodule add -\/-\/force git@github.com:DoxygenGeneratorWindow/DoxygenGeneratorWindow.git}
\end{DoxyCode}


\doxysubsection*{Download Doxygen}


\begin{DoxyItemize}
\item go to \href{https://www.doxygen.nl}{\texttt{ https\+://www.\+doxygen.\+nl}}
\item download Doxygen© for your system
\item install Doxygen©
\end{DoxyItemize}

\doxysubsection*{Launch your project}


\begin{DoxyItemize}
\item select Doxygen Generator Window in menu
\end{DoxyItemize}




\begin{DoxyItemize}
\item select path of Doxygen (you just downloaded it...)
\item select path of Folder sources (your source code...)
\item select destination path 
\item fill the form
\item select your options 
\item click on \char`\"{}\+Generate\char`\"{}
\item enjoy your documentation 
\end{DoxyItemize}




\begin{DoxyItemize}
\item generate your code
\end{DoxyItemize}
